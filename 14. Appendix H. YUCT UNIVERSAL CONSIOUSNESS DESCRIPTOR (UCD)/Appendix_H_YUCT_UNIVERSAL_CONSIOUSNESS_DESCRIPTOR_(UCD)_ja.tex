		def correlation_dimension(data, r_min=0.01, r_max=1.0, n_r=20):
		r_vals = np.logspace(np.log10(r_min), np.log10(r_max), n_r)

		while k + l <= n:
		if binary_string[i:i+l] == binary_string[k:k+l]:
		l += 1
		else:
		i += 1
		if i == k:
		complexity += 1
		k = k + l
		l = 1
		else:
